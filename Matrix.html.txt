    <script>
        function addTask() {
            const name = document.getElementById('taskName').value;
            const importance = parseInt(document.getElementById('importance').value);
            const urgency = parseInt(document.getElementById('urgency').value);

            if (!name) {
                alert("Please enter a task description.");
                return;
            }

            // Determine Quadrant based on weights (Threshold is 6)
            let targetDiv = '';
            if (importance >= 6 && urgency >= 6) {
                targetDiv = 'tasks-q1'; // Q1: Do First
            } else if (importance >= 6 && urgency < 6) {
                targetDiv = 'tasks-q2'; // Q2: Schedule
            } else if (importance < 6 && urgency >= 6) {
                targetDiv = 'tasks-q3'; // Q3: Delegate
            } else {
                targetDiv = 'tasks-q4'; // Q4: Eliminate
            }

            // Create Task Element
            const taskEl = document.createElement('div');
            taskEl.className = 'task-item';
            
            // Generate unique ID for deletion
            const taskId = 'task-' + Date.now();
            taskEl.id = taskId;

            // Set the task's inner HTML
            taskEl.innerHTML = `
                <div class="task-details">
                    <strong>${name}</strong>
                    <span class="task-weight">Importance: ${importance}/10 | Urgency: ${urgency}/10</span>
                </div>
                <button class="delete-btn" onclick="deleteTask('${taskId}')">X</button>
            `;

            // Append task to the calculated quadrant
            document.getElementById(targetDiv).appendChild(taskEl);

            // Clear input for the next task
            document.getElementById('taskName').value = '';
            document.getElementById('taskName').focus();
        }

        function deleteTask(id) {
            document.getElementById(id).remove();
        }
    </script>
